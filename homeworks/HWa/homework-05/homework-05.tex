% Homework problem 2 sec 3.1 problem 6
\begin{homeworkProblem}
	Let $A$ be an $m \times n$ matrix. Prove that if $B$ can be obtained from $A$ by an elementary row [column] operation, then $B^T$ can be obtained from $A^T$ by the
	corresponding elementary column [row] operation.
	\begin{proof}
		\begin{align*}
			\text{Let } E_i \text{ be the elementary row operation that transforms } A \text{ into } B        \\
			B   & = E_i \cdots E_2 E_1 A                                                                      \\
			B^T & = (E_i \cdots E_2 E_1 A)^T                                                                  \\
			B^T & = A E_1^T E_2^T \cdots E_i^T                                                                \\
			\text{Let } E_j \text{ be the elementary column operation that transforms } A^T \text{ into } B^T \\
			B   & = A E_1 E_2 \cdots E_j                                                                      \\
			B^T & = (A E_1 E_2 \cdots E_j)^T                                                                  \\
			B^T & = E_j^T \cdots E_2^T E_1^T A^T
		\end{align*}
		Therefore, $B^T$ can be obtained from $A^T$ by the corresponding elementary column operation.
	\end{proof}
\end{homeworkProblem}

% Homework problem 3 sec 3.1 problem 10
\begin{homeworkProblem}
	Prove that any elementary row [column] operation of type 2 can be obtained by dividing some row [column] by a nonzero scalar.
	\begin{proof}
		Let $A$ be an $m \times n$ matrix and $B$ be the matrix obtained by dividing the $i$-th row of $A$ by a nonzero scalar $c$ which would be equivalent to multiplying the
		$i$-th row of $A$ by some scalar $k$
		\begin{align*}
			c = \frac{1}{\frac{1}{k}} \\
			c = k
		\end{align*}
		Therefore, the elementary row operation of type 2 can be obtained by dividing some row by a nonzero scalar.
	\end{proof}
\end{homeworkProblem}

% Homework problem 6 sec 3.2 problem 6b
\begin{homeworkProblem}
	For each of the following linear transformations $T$, determine whether $T$ is invertible, and compute $T^{-1}$ if it exists:
	$$ T = P_2 (\mathbb{R}) \to P_2 (\mathbb{R}) \quad \text{defined by} \quad T(f(x)) = (x+1)f'(x)$$
	\[
		\begin{split}
			\text{Let } f'(x) = 2ax + b \\
			\text{Then } T(f(x)) = (x+1)(2ax + b) = 2ax^2 + (2a + b)x + b \\
			\text{Let } g(x) = 2dx + e \\
			f(x) = ax^2 + bx + c \text{ and } g(x) = dx^2 + ex + f \\
		\end{split}
	\]
	Since $f(x)$ is constant when $a = 0$ and $b = 0$, then $T$ is not injective. Therefore, $T$ is not invertible.
\end{homeworkProblem}

% Homework problem 8 sec 3.2 problem 14
\begin{homeworkProblem}
	Let $T, U: V \to W$ be linear transformations.
	\begin{itemize}
		\item Prove that $R(T + U) \subseteq R(T) + R(U)$.
		      \[
			      \begin{split}
				      \text{Let } v \in V \\
				      (T + U)(v) = T(v) + U(v) \\
				      \text{ Since } T(v) \in R(T) \text{ and } U(v) \in R(U) \\
				      T(v) + U(v) \in R(T) + R(U) \\
				      R(T + U) \subseteq R(T) + R(U)
			      \end{split}
		      \]
		\item Prove that if $W$ if finite-dimensional, then $rank(T + U) \leq rank(T) + rank(U)$.
		      \[
			      \begin{split}
				      R(T + U) & \subseteq R(T) + R(U) \\
				      dim(R(T + U)) & \leq dim(R(T) + R(U)) \\
				      dim(R(T + U)) & \leq dim(R(T)) + dim(R(U)) \\
				      rank(T + U) & \leq rank(T) + rank(U)
			      \end{split}
		      \]
		\item Deduce from (b) that if $rank(A + B) \leq rank(A) + rank(B)$ for any $m \times n$ matrices $A$ and $B$.
	\end{itemize}
\end{homeworkProblem}
